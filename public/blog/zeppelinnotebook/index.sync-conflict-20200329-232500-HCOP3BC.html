Install Zeppelin

cd zeppelin
wget http://miroir.univ-lorraine.fr/apache/zeppelin/zeppelin-0.8.2/zeppelin-0.8.2-bin-all.tgz

sudo tar -zxf zeppelin-0.8.2-bin-all.tgz

Run Zeppelin Server

cd zeppelin-0.8.2-bin-all
sudo bin/zeppelin-daemon.sh start
